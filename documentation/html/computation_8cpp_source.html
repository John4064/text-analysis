54void *task(void *rec_struct) {
55
56 auto *struct_ptr = (struct inputStruct*) rec_struct;
57 //How large the thread processes ex 4 threads of 100 lines do 25 each
58 unsigned int size =struct_ptr->bookP->size()/struct_ptr->numOfThread;
59 unsigned int ind = struct_ptr->index;
60 //The start index of the pages to analyze
61 unsigned int startInd = size*ind;
62 //The vector the book is stored in & and word frequency map
63 std::vector<std::string> bookVec = *struct_ptr->bookP;
64 std::map<std::string, int>* wordL = struct_ptr->wordMap;
65 //Iterate through our subsection!
66 for(unsigned int i =startInd; i < startInd+size; i++){
67 //We split each string by space then put into vector based on frequency
68 //This is the string stream of our line
69 std::stringstream ss(bookVec[i]);
70 //Each individual word
71 std::string inWord;
72 while (ss>> inWord) {
73 //Preprocess inWord For punctuation
74 inWord=cleanup(inWord);
75 if(inWord.empty()){
76 //Skips just empty
77 continue;
78 }
79 //Determine if it is in vector
80 freqMtx.lock();
81 std::map<std::string, int>::iterator it = wordL->find(inWord);
82 if(it != wordL->end()) {
83 //Update
84 it->second = it->second+1;
85 }else{
86 //Doesnt Exist so insert into map
87 wordL->insert(std::make_pair(inWord,1));
88 }
89 freqMtx.unlock();
90 }
91 }
92 pthread_exit(nullptr);
93 return nullptr;
94}
